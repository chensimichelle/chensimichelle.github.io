let score = 0;
let answered = 0;
const total = questions.length;

function updateScore() {
  document.getElementById('scoreDisplay').textContent = `${score} / ${total}`;
  const pct = (answered / total) * 100;
  document.getElementById('progressBar').style.width = pct + '%';
  if (answered === total) {
    document.getElementById('summary').classList.add('show');
    document.getElementById('summaryScore').textContent = `${score}/${total}`;
    const pctScore = Math.round((score / total) * 100);
    document.getElementById('summaryMsg').textContent =
      pctScore >= 80 ? "Excellent work! 🎉" :
      pctScore >= 60 ? "Good effort — review the ones you missed." :
      "Keep practising — biology takes time!";
  }
}

function buildQuiz() {
  const container = document.getElementById('quizContainer');

  questions.forEach((q, idx) => {
    const card = document.createElement('div');
    card.className = 'question-card';
    card.id = `card-${idx}`;

    const header = document.createElement('div');
    header.className = 'question-header';

    const num = document.createElement('span');
    num.className = 'q-number';
    num.textContent = q.id;

    const text = document.createElement('div');
    text.className = 'q-text';
    text.innerHTML = q.text.replace(/\n/g, '<br>');

    header.appendChild(num);
    header.appendChild(text);
    card.appendChild(header);

    const opts = document.createElement('div');
    opts.className = 'options';

    const feedback = document.createElement('div');
    feedback.className = 'feedback';

    q.options.forEach(opt => {
      const letter = opt.split(':')[0].trim();
      const label = opt.substring(opt.indexOf(':') + 1).trim();
      const btn = document.createElement('button');
      btn.className = 'option';
      btn.dataset.letter = letter;
      btn.innerHTML = `<span class="option-label">${letter}</span>${label}<span class="option-icon"></span>`;
      btn.addEventListener('click', () => handleAnswer(idx, letter, btn, opts, feedback, card));
      opts.appendChild(btn);
    });

    card.appendChild(opts);
    card.appendChild(feedback);
    container.appendChild(card);
  });

  const summary = document.createElement('div');
  summary.className = 'summary';
  summary.id = 'summary';
  summary.innerHTML = `
    <p style="font-family:'DM Mono',monospace;font-size:0.75rem;color:var(--muted);letter-spacing:0.1em;text-transform:uppercase">Quiz Complete</p>
    <div class="big-score" id="summaryScore">0/20</div>
    <h2>Final Score</h2>
    <p id="summaryMsg"></p>
    <button class="reset-btn" onclick="location.reload()">Try Again</button>
  `;
  container.appendChild(summary);
}

function handleAnswer(idx, chosen, btn, opts, feedback, card) {
  const q = questions[idx];
  const correct = q.answer;
  const allBtns = opts.querySelectorAll('.option');

  allBtns.forEach(b => b.disabled = true);

  if (chosen === correct) {
    btn.classList.add('correct');
    btn.querySelector('.option-icon').textContent = '✓';
    card.classList.add('answered-correct');
    feedback.className = 'feedback correct-fb show';
    feedback.innerHTML = `<span>✓</span> Correct! ${q.feedback}`;
    score++;
  } else {
    btn.classList.add('wrong');
    btn.querySelector('.option-icon').textContent = '✗';
    card.classList.add('answered-wrong');
    allBtns.forEach(b => {
      if (b.dataset.letter === correct) b.classList.add('show-correct');
    });
    feedback.className = 'feedback wrong-fb show';
    feedback.innerHTML = `<span>✗</span> Incorrect. The answer is <strong>${correct}</strong>. ${q.feedback}`;
  }

  answered++;
  updateScore();
}

buildQuiz();
